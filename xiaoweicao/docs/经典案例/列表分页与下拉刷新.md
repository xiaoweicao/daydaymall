### 列表分页与下拉刷新（可直接复制）

```vue
<template>
  <scroll-view scroll-y style="height:100vh" @scrolltolower="loadMore" @refresherrefresh="onRefresh" refresher-enabled>
    <view v-for="item in list" :key="item.id" class="row">{{ item.name }}</view>
    <view v-if="loading" class="loading">加载中...</view>
    <view v-if="finished" class="end">没有更多了</view>
  </scroll-view>
  <view class="refresh" @click="onRefresh">手动刷新</view>
  </template>

<script setup>
import { ref } from 'vue'
const page = ref(1)
const pageSize = 20
const list = ref([])
const loading = ref(false)
const finished = ref(false)

async function fetchList(reset = false) {
  if (loading.value || finished.value) return
  loading.value = true
  if (reset) { page.value = 1; finished.value = false }
  const data = await mockApi(page.value, pageSize)
  list.value = page.value === 1 ? data : [...list.value, ...data]
  if (data.length < pageSize) finished.value = true
  page.value++
  loading.value = false
}

function loadMore() { fetchList() }
function onRefresh() { fetchList(true) }

function mockApi(p, ps) {
  return new Promise(resolve => setTimeout(() => {
    const start = (p - 1) * ps
    resolve(Array.from({ length: p < 5 ? ps : 0 }, (_, i) => ({ id: start + i, name: 'Item ' + (start + i) })))
  }, 300))
}

fetchList(true)
</script>

<style scoped>
.row { padding: 24rpx; border-bottom: 1px solid #eee }
.loading, .end { text-align: center; padding: 24rpx; color: #888 }
.refresh { position: fixed; right: 24rpx; bottom: 24rpx; background: #07c160; color: #fff; padding: 16rpx 24rpx; border-radius: 12rpx }
</style>
```
